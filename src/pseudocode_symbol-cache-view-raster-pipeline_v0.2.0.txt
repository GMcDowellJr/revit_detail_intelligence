# ============================================================
# SYMBOL CACHE + VIEW RASTER PIPELINE  v0.2.0
#
# Scope:
#   - Per-symbol raster cache for geometry recovery on opaque detail items
#   - Full-view raster cache for optional re-ranking
#   - Two-stage pipeline integration with existing feature-based search
#
# Changes from v0.1.0:
#   - SymbolKey is name-based (family_name|type_name), not numeric ID
#   - Family document lifecycle uses try/finally; never leaks open documents
#   - FamilySymbol activation is explicit, transacted, and failure-logged
#   - All write operations are wrapped in scoped transactions with rollback
#   - ViewExportConfig is versioned and hashed for reproducibility
#   - Connected-component filtering uses configurable threshold policy,
#     not hard "keep largest" — preserves multi-body detail geometry
#   - Descriptor aggregation is count-weighted, not a simple mean
#   - Symbol cache invalidates on three axes: identity, content, infra version
#   - View raster cache key derived from ViewFeatures already in the pipeline
#   - Two-stage integration: score_total (feature layer) is never mutated;
#     score_combined drives re-rank ordering only when raster data is present
#   - Cache storage layout, lifecycle policy, and sharing rules defined
# ============================================================

# --- Symbol identity ---

TYPE SymbolKey:
  family_name: str
  type_name: str

  # Numeric IDs are retained for Revit API lookups ONLY during cache build.
  # They must never be used as cache keys or stored as identity.
  # They are not stable across file save/reload cycles.
  _revit_family_id:  int | null   # lookup hint only; not serialized
  _revit_symbol_id:  int | null   # lookup hint only; not serialized

  function stable_string() -> str:
    # Matches type_signature() convention in dse/features/tokens.py
    return "{family_name}|{type_name}"

  function from_element(element) -> SymbolKey:
    sig = type_signature(element)   # reuse existing tokens.py function
    parts = sig.split("|", 1)
    return SymbolKey(
      family_name      = parts[0],
      type_name        = parts[1] if len(parts) > 1 else "",
      _revit_family_id = element.Symbol.Family.Id.IntegerValue if has symbol else null,
      _revit_symbol_id = element.Symbol.Id.IntegerValue if has symbol else null,
    )


# --- Symbol descriptor ---

TYPE SymbolDescriptor:
  schema: str                  # "symbol_descriptor.v2"
  cache_key: str               # SymbolKey.stable_string()
  validity_token: str          # recomputed on lookup; mismatch = rebuild
  build_method: str            # "family_doc" | "isolated_render"
  build_time_utc: str
  export_config_hash: str      # ViewExportConfig.content_hash(); mismatch = rebuild

  source_fingerprint: dict     # provenance for auditability (not used for invalidation)
    family_source_path: str | null
    param_hash: str

  # Primary matching descriptor
  descriptor_vector: list[float]
  descriptor_vector_mirror: list[float]   # flip_x variant; for mirror-invariant comparison
  descriptor_kind: str

  # Auxiliary signals
  bbox_aspect: float
  bbox_area_norm: float
  pixel_density: float
  debug_preview_path: str | null

TYPE SymbolCache:

  descriptors: dict[str, SymbolDescriptor]  # key = SymbolKey.stable_string()

# SECTION 3: CACHE KEYS AND VALIDITY TOKENS
# ============================================================

# --- Symbol validity token ---
# Three independent invalidation axes are checked on every lookup:
#   (1) Identity:      cache_key = family_name|type_name
#   (2) Content:       param_hash — detects type parameter changes
#   (3) Infrastructure: export_config_hash — detects settings version changes
#
# If any axis fails, the entry is treated as a miss and rebuilt.
# The rebuild reason should be logged for diagnostics.

function symbol_type_parameter_hash(family_symbol) -> str:
  params = []
  for param in family_symbol.Parameters:
    try:
      name    = param.Definition.Name
      value   = param.AsValueString() or param.AsString() or ""
      storage = str(param.StorageType)
      params.append((name, storage, value))
    except:
      continue   # skip unreadable params; not a fatal error

  params.sort(by=name)
  canonical = json_serialize(params)
  return sha256(canonical)[0:16]

  # NOTE: Does not detect purely graphical changes (line weight, fill pattern)
  # made inside the family editor without a type parameter change.
  # Accepted limitation: those changes do not affect structural geometry
  # captured by the descriptor.


function build_symbol_validity_token(family_symbol, export_config) -> str:
  cache_key   = SymbolKey.from_element(family_symbol).stable_string()
  param_hash  = symbol_type_parameter_hash(family_symbol)
  config_hash = export_config.content_hash()
  combined    = "{cache_key}::{param_hash}::{config_hash}"
  return sha256(combined)[0:20]


function get_cached_symbol_descriptor(cache, family_symbol, export_config) -> SymbolDescriptor | null:
  key   = SymbolKey.from_element(family_symbol).stable_string()
  entry = cache.descriptors.get(key)
  if entry is null:
    log_cache_miss(key, reason="not found")
    return null

  current_token = build_symbol_validity_token(family_symbol, export_config)
  if entry.validity_token != current_token:
    log_cache_miss(key, reason="validity_token mismatch — content or config changed")
    return null

  return entry


# --- View raster cache key ---
# Derived from ViewFeatures already computed in the main pipeline.
# No additional feature extraction is needed to check the cache.
#
# If tokens or geometry fingerprint change, hash changes and raster is rebuilt.
# Does not capture visibility overrides or graphical changes that don't affect
# geometry — a manual cache-clear is sufficient for those rare cases.

function view_features_hash(view_features) -> str:
  token_repr = sorted(view_features.tokens.items())
  geom_repr  = [round(v, 8) for v in view_features.geom_fingerprint]
  content = {
    "tokens":    token_repr,
    "geom":      geom_repr,
    "view_kind": view_features.view_kind,
  }
  canonical = json_serialize(content)
  return sha256(canonical)[0:20]


function build_view_raster_cache_key(view_features, export_config_hash) -> str:
  content = "{view_features.view_id}::{view_features_hash(view_features)}::{export_config_hash}"
  return sha256(content)[0:24]


function get_cached_view_raster(raster_cache, view_features, export_config) -> ViewRasterSignature | null:
  key   = build_view_raster_cache_key(view_features, export_config.content_hash())
  entry = raster_cache.get(key)
  if entry is null:
    log_cache_miss(view_features.view_id, reason="not found")
    return null
  return entry

# All fields that affect pixel content are pinned here.
# Increment SCHEMA_VERSION whenever any default changes.
# The content hash is stored in every descriptor and checked on lookup.

TYPE ViewExportConfig:
  SCHEMA_VERSION:   str = "export_config.v1"
  pixel_size:       int = 512
  background_color: str = "white"
  export_range:     str = "visible"
  zoom_type:        str = "fitpage"
  antialiasing:     bool = True

  function to_revit_options():
    opts = ImageExportOptions()
    opts.PixelSize  = self.pixel_size
    opts.ZoomType   = ZoomFitType.FitPage
    # ... map remaining fields to Revit API types
    return opts

  function content_hash() -> str:
    fields = {
      "_schema":          self.SCHEMA_VERSION,
      "pixel_size":       self.pixel_size,
      "background_color": self.background_color,
      "export_range":     self.export_range,
      "zoom_type":        self.zoom_type,
      "antialiasing":     self.antialiasing,
    }
    canonical = json_serialize(fields, sort_keys=True)
    return sha256(canonical)[0:16]


# ============================================================
# SECTION 5: TRANSACTION WRAPPER
# ============================================================

# Rule: every Revit write operation (view create, symbol place, view delete)
# runs inside its own scoped_transaction. Transactions are never nested.
# Export is a read-only side effect and runs outside any transaction.
# Failure in any transaction triggers rollback and surfaces a Failure value
# to the caller — never swallowed silently.

context_manager scoped_transaction(doc, name):
  t = Transaction(doc, name)
  t.Start()
  try:
    yield t
    status = t.Commit()
    if status != TransactionStatus.Committed:
      raise RuntimeError("transaction '{name}' did not commit: {status}")
  except any_exception:
    try:
      t.RollBack()
    except rollback_exception:
      log_warning("rollback failed for '{name}': {rollback_exception}")
    raise


# ============================================================
# SECTION 6: FAMILY DOCUMENT LIFECYCLE
# ============================================================

# Rule: family documents are always closed in a finally block.
# The context manager guarantees closure even if extraction raises.
# Close is always called with discard=True (no changes are made).

context_manager open_family_doc(project_doc, family):
  fam_doc = null
  try:
    fam_doc = project_doc.EditFamily(family)
    if fam_doc is null:
      raise RuntimeError("EditFamily returned null for: {family.Name}")
    yield fam_doc
  finally:
    if fam_doc is not null:
      try:
        fam_doc.Close(discard=True)
      except close_exception:
        log_warning("failed to close family doc '{family.Name}': {close_exception}")
        # Do not re-raise: leaking a warning is better than masking the original error.


# ============================================================
# SECTION 7: SYMBOL ACTIVATION
# ============================================================

# Rule: activation is always called inside an open transaction.
# Failure raises with context; caller decides skip vs. abort policy.
# Revit 2025 requires FamilySymbol.Activate() before placement
# if the symbol has not been previously used in the document.

function ensure_symbol_activated(symbol, doc):
  # Must be called inside scoped_transaction.
  try:
    if not symbol.IsActive:
      symbol.Activate()
      doc.Regenerate()
  except activation_exception:
    raise RuntimeError(
      "failed to activate symbol '{symbol.Name}': {activation_exception}"
    )

# SECTION 8: IMAGE NORMALIZATION
# ============================================================

function raster_to_binary_or_edges(png_path, preprocess_config) -> Image2D:
  img      = load_png(png_path)
  img_gray = to_grayscale(img)
  if preprocess_config.mode == "edges":
    edge = edge_detect(img_gray, preprocess_config.edge_params)
    return threshold(edge, preprocess_config.edge_threshold)
  else:
    return adaptive_threshold(img_gray, preprocess_config.thresh_params)


function filter_connected_components(bin_img, policy, min_area_ratio) -> Image2D:
  # policy = "largest"   — keep only the single largest CC (lossy; not recommended
  #                        for detail items with legitimate multi-body geometry)
  # policy = "threshold" — keep all CCs with area >= min_area_ratio * largest CC area
  #                        (recommended default: removes rasterization noise while
  #                        preserving steel angles, insulation outline+fill, etc.)
  # policy = "all"       — no filtering

  components = find_connected_components(bin_img)
  if components is empty:
    return bin_img

  max_area = max(c.area for c in components)

  if policy == "largest":
    keep = [c for c in components if c.area == max_area]
  elif policy == "threshold":
    keep = [c for c in components if c.area >= min_area_ratio * max_area]
  else:  # "all"
    keep = components

  return reconstruct_image_from_components(keep, bin_img.shape)


function normalize_image(bin_img, output_px, pad_px, cc_filter_policy, cc_min_area_ratio) -> Image2D:
  bin_img  = filter_connected_components(bin_img, cc_filter_policy, cc_min_area_ratio)
  bbox     = bounding_box_of_foreground(bin_img)
  cropped  = crop(bin_img, bbox)
  padded   = pad_to_square(cropped, pad_px)
  resized  = resize(padded, output_px, output_px)
  return recenter_foreground(resized)


function mirror_descriptor_strategy(bin_img, descriptor_kind) -> (list[float], list[float]):
  D  = compute_rotation_invariant_descriptor(bin_img, descriptor_kind)
  Dm = compute_rotation_invariant_descriptor(flip_x(bin_img), descriptor_kind)
  return (D, Dm)


function compute_rotation_invariant_descriptor(bin_img, descriptor_kind) -> list[float]:
  if descriptor_kind == "zernike_mag":
    return zernike_magnitudes(bin_img)
  if descriptor_kind == "fourier_mag":
    contour         = extract_outer_contour(bin_img)
    contour_samples = resample_contour_uniform(contour, n=256)
    F               = dft(contour_samples)
    return magnitudes(F[1:K])
  if descriptor_kind == "edge_hist":
    return oriented_edge_histogram(bin_img)
  raise ValueError("unknown descriptor_kind: {descriptor_kind}")


# ============================================================
# SECTION 9: DESCRIPTOR AGGREGATION
# ============================================================

# Count-weighted aggregation: a view with 3 instances of symbol A and 1 of
# symbol B produces a weighted mean with A contributing 3x, not a simple
# mean of unique descriptors. This matches the weighted Jaccard approach
# used in the token layer.

function aggregate_descriptors(descriptor_vectors, weights) -> list[float]:
  # weights: per-instance counts or importances (default: uniform)
  if descriptor_vectors is empty:
    return []

  weights      = weights or [1.0] * len(descriptor_vectors)
  total_weight = sum(weights)
  if total_weight <= EPS:
    return [0.0] * len(descriptor_vectors[0])

  dim    = len(descriptor_vectors[0])
  result = [0.0] * dim
  for (vec, w) in zip(descriptor_vectors, weights):
    for i in range(dim):
      result[i] += vec[i] * w

  norm = sqrt(sum(x*x for x in result))
  if norm <= EPS:
    return result
  return [x / norm for x in result]


function expand_symbol_multiset(symbol_counts, symbol_cache) -> (list, list, list):
  # symbol_counts: dict of {cache_key_str: count}
  # Returns: (descriptor_vectors, weights, missing_keys)
  vectors  = []
  weights  = []
  missing  = []
  for (key_str, count) in symbol_counts.items():
    desc = symbol_cache.descriptors.get(key_str)
    if desc is null:
      missing.append(key_str)
      continue
    vectors.append(desc.descriptor_vector)
    weights.append(float(count))
  return (vectors, weights, missing)

# SECTION 10: SIMILARITY FUNCTIONS
# ============================================================

function descriptor_similarity(vec_a, vec_a_mirror, vec_b, vec_b_mirror) -> float:
  # Mirror-invariant: take best of normal and mirrored comparison.
  sim1 = cosine_similarity(vec_a, vec_b)
  sim2 = cosine_similarity(vec_a, vec_b_mirror)
  return max(sim1, sim2)


function cosine_similarity(a, b) -> float:
  if a is empty or b is empty or len(a) != len(b):
    return 0.0
  dot = sum(x*y for (x,y) in zip(a,b))
  na  = sqrt(sum(x*x for x in a))
  nb  = sqrt(sum(y*y for y in b))
  if na <= EPS or nb <= EPS:
    return 0.0
  return dot / (na * nb)

# SECTION 11: SYMBOL CACHE BUILD — MODE A: FAMILY-DOC EXTRACTION
# ============================================================

# Best-effort fast path. Opens the family document and extracts symbolic
# lines and region boundaries directly. Falls back to isolated render if
# extraction yields no primitives.
#
# Document lifecycle is guaranteed by open_family_doc context manager.
# No transactions required — this is read-only against the project document.

function build_symbol_descriptor_family_doc(project_doc, symbol_key, config) -> SymbolDescriptor | Failure:
  family = resolve_family_by_name(project_doc, symbol_key.family_name)
  if family is null:
    return Failure("family not found: {symbol_key.stable_string()}")

  try:
    with open_family_doc(project_doc, family) as fam_doc:
      curves  = []
      regions = []

      for elem in fam_doc.elements:
        if elem is symbolic_line_or_curve:
          curves.extend(curves_from_symbolic_line(elem))
        if elem is masking_region_or_filled_region:
          regions.append(region_boundary_curves(elem))

      if curves is empty and regions is empty:
        return Failure("no extractable primitives in family doc: {symbol_key.stable_string()}")

      return _build_descriptor_from_primitives(
        curves, regions, symbol_key, config, build_method="family_doc"
      )

  except extraction_exception:
    return Failure("family_doc extraction failed for {symbol_key.stable_string()}: {extraction_exception}")


# ============================================================
# SECTION 12: SYMBOL CACHE BUILD — MODE B: ISOLATED RENDER
# ============================================================

# Robust fallback. Places a single symbol instance in a temporary drafting
# view, exports a PNG, then deletes the view. Each write operation is its own
# transaction so cleanup runs even if export fails.
#
# Transaction ownership: Dynamo manages the outer document context.
# Each scoped_transaction here is a leaf transaction (no nesting).

function build_symbol_descriptor_isolated_render(project_doc, symbol_key, config) -> SymbolDescriptor | Failure:
  # Resolve the FamilySymbol via name (stable) not stored numeric ID.
  symbol = resolve_family_symbol_by_name(
    project_doc, symbol_key.family_name, symbol_key.type_name
  )
  if symbol is null:
    return Failure("symbol not found: {symbol_key.stable_string()}")

  # --- create temp view (write transaction) ---
  tmp_view = null
  try:
    with scoped_transaction(project_doc, "DSE: create temp view") as _:
      tmp_view = ViewDrafting.Create(project_doc, config.view_family_type_id)
  except transaction_exception:
    return Failure("temp view create failed: {transaction_exception}")

  # --- activate symbol and place instance (write transaction) ---
  inst = null
  try:
    with scoped_transaction(project_doc, "DSE: place symbol instance") as _:
      ensure_symbol_activated(symbol, project_doc)   # raises on failure
      inst = project_doc.Create.NewFamilyInstance(XYZ.Zero, symbol, tmp_view)
  except transaction_exception:
    _delete_view_safe(project_doc, tmp_view)
    return Failure("symbol placement failed for {symbol_key.stable_string()}: {transaction_exception}")

  # --- export PNG (read-only side effect; outside any transaction) ---
  isolate_elements_temporarily(tmp_view, [inst.Id])
  crop_bbox = element_bbox_in_view(inst, tmp_view) or default_crop_box(config.default_crop_size)
  set_view_crop_box(tmp_view, expand_bbox(crop_bbox, config.crop_padding_ratio))

  png_path = export_view_png(project_doc, tmp_view, config.export_config.to_revit_options())

  # --- delete temp view regardless of export outcome (write transaction) ---
  _delete_view_safe(project_doc, tmp_view)

  if png_path is null or not file_exists(png_path):
    return Failure("png export failed for {symbol_key.stable_string()}")

  # --- build descriptor from exported PNG ---
  try:
    return _build_descriptor_from_png(png_path, symbol_key, config, build_method="isolated_render")
  except descriptor_exception:
    return Failure("descriptor build failed: {descriptor_exception}")
  finally:
    delete_file_if_exists(png_path)   # always clean up temp PNG


function _delete_view_safe(doc, view):
  if view is null:
    return
  try:
    with scoped_transaction(doc, "DSE: delete temp view") as _:
      doc.Delete(view.Id)
  except transaction_exception:
    log_warning("failed to delete temp view {view.Id}: {transaction_exception}")
    # Do not re-raise: cleanup failure should not mask the original error.

# SECTION 13: SHARED DESCRIPTOR BUILD HELPERS
# ============================================================

function _build_descriptor_from_primitives(curves, regions, symbol_key, config, build_method) -> SymbolDescriptor:
  bin_img   = rasterize_curves_and_regions_to_binary(curves, regions, config.rasterize_px)
  bin_img_n = normalize_image(
    bin_img,
    config.image_normalize.output_px,
    config.image_normalize.pad_px,
    config.image_normalize.cc_filter_policy,
    config.image_normalize.cc_min_area_ratio,
  )
  (D, Dm) = mirror_descriptor_strategy(bin_img_n, config.image_normalize.descriptor_kind)
  return _assemble_descriptor(D, Dm, bin_img_n, symbol_key, config, build_method)


function _build_descriptor_from_png(png_path, symbol_key, config, build_method) -> SymbolDescriptor:
  bin_img   = raster_to_binary_or_edges(png_path, config.preprocess)
  bin_img_n = normalize_image(
    bin_img,
    config.image_normalize.output_px,
    config.image_normalize.pad_px,
    config.image_normalize.cc_filter_policy,
    config.image_normalize.cc_min_area_ratio,
  )
  (D, Dm) = mirror_descriptor_strategy(bin_img_n, config.image_normalize.descriptor_kind)
  return _assemble_descriptor(D, Dm, bin_img_n, symbol_key, config, build_method)


function _assemble_descriptor(D, Dm, bin_img_n, symbol_key, config, build_method) -> SymbolDescriptor:
  return SymbolDescriptor(
    schema                    = "symbol_descriptor.v2",
    cache_key                 = symbol_key.stable_string(),
    validity_token            = null,            # set by caller after build
    build_method              = build_method,
    build_time_utc            = utc_now(),
    export_config_hash        = config.export_config.content_hash(),
    source_fingerprint        = {
      "mode":           build_method,
      "output_px":      config.image_normalize.output_px,
      "descriptor_kind":config.image_normalize.descriptor_kind,
    },
    descriptor_vector         = D,
    descriptor_vector_mirror  = Dm,
    descriptor_kind           = config.image_normalize.descriptor_kind,
    bbox_aspect               = aspect_of_foreground(bin_img_n),
    bbox_area_norm            = area_of_foreground(bin_img_n) / (config.image_normalize.output_px ** 2),
    pixel_density             = foreground_density(bin_img_n),
    debug_preview_path        = maybe_write_preview(bin_img_n, symbol_key, config),
  )

# SECTION 14: SYMBOL CACHE ORCHESTRATION
# ============================================================

function build_symbol_cache(project_doc, symbol_keys, config) -> SymbolCache:
  cache    = load_or_create_symbol_cache(config.symbol_cache_policy.storage_path_pattern)
  failures = []

  for key in symbol_keys:
    # Resolve the actual FamilySymbol for validity token computation.
    symbol = resolve_family_symbol_by_name(project_doc, key.family_name, key.type_name)
    if symbol is null:
      failures.append({"symbol": key.stable_string(), "reason": "symbol not found in document"})
      continue

    # Check all three invalidation axes before deciding to rebuild.
    existing = get_cached_symbol_descriptor(cache, symbol, config.export_config)
    if existing is not null and not config.force_rebuild:
      continue   # valid cache hit on all three axes

    # Mode A: family-doc extraction (fast, explainable)
    desc = build_symbol_descriptor_family_doc(project_doc, key, config)

    # Mode B: isolated render (robust fallback; sees what the user sees)
    if desc is Failure:
      log_info("falling back to isolated_render for {key.stable_string()}: {desc.reason}")
      desc = build_symbol_descriptor_isolated_render(project_doc, key, config)

    if desc is Failure:
      failures.append({"symbol": key.stable_string(), "reason": desc.reason})
      continue

    # Attach validity token now that we know the symbol and config.
    desc.validity_token = build_symbol_validity_token(symbol, config.export_config)

    cache.descriptors[key.stable_string()] = desc

  cache.stats = {
    "symbols_total":   len(symbol_keys),
    "symbols_cached":  len(cache.descriptors),
    "failures":        len(failures),
    "failure_samples": failures[0:20],
  }
  write_json(cache_path_for_project(config, project_doc), cache)
  return cache

# SECTION 15: VIEW RASTER PIPELINE
# ============================================================

# Each write operation is its own transaction.
# Export runs outside any transaction (read-only side effect).
# The temp view is always deleted, even if export fails.

function render_view_png_standard(project_doc, view, config) -> str | Failure:
  # Duplicate view to avoid mutating the user's view.
  tmp_view = null
  try:
    with scoped_transaction(project_doc, "DSE: duplicate view for export") as _:
      tmp_view = duplicate_view_for_export(view)
      apply_view_template(tmp_view, config.view_template_id)
  except transaction_exception:
    return Failure("view duplication failed: {transaction_exception}")

  # Apply crop strategy (still inside a transaction if it requires writes).
  try:
    with scoped_transaction(project_doc, "DSE: set view crop") as _:
      if config.crop_strategy == "content_bbox":
        elems = collect_elements_for_crop(tmp_view, include_categories=config.crop_categories)
        bbox  = union_bbox([element_bbox_in_view(e, tmp_view) for e in elems])
        bbox  = bbox or fallback_crop_box(tmp_view)
        set_view_crop_box(tmp_view, expand_bbox(bbox, config.crop_padding_ratio))
      elif config.crop_strategy == "view_cropbox":
        ensure_crop_enabled(tmp_view)
      elif config.crop_strategy == "fixed_frame":
        set_view_crop_box(tmp_view, fixed_box(config.fixed_frame_size))
      else:
        raise ValueError("unknown crop_strategy: {config.crop_strategy}")
  except transaction_exception:
    _delete_view_safe(project_doc, tmp_view)
    return Failure("crop setup failed: {transaction_exception}")

  # Export outside any transaction.
  png_path = export_view_png(project_doc, tmp_view, config.export_config.to_revit_options())

  _delete_view_safe(project_doc, tmp_view)

  if png_path is null or not file_exists(png_path):
    return Failure("png export failed for view {view.Id}")

  return png_path


function build_view_raster_signature(project_doc, view, view_features, config) -> ViewRasterSignature | Failure:
  png_path = render_view_png_standard(project_doc, view, config.render)
  if png_path is Failure:
    return png_path

  try:
    bin_img   = raster_to_binary_or_edges(png_path, config.preprocess)
    bin_img_n = normalize_image(
      bin_img,
      config.image_normalize.output_px,
      config.image_normalize.pad_px,
      config.image_normalize.cc_filter_policy,
      config.image_normalize.cc_min_area_ratio,
    )
    (D, Dm) = mirror_descriptor_strategy(bin_img_n, config.image_normalize.descriptor_kind)
  except processing_exception:
    return Failure("image processing failed: {processing_exception}")
  finally:
    delete_file_if_exists(png_path)

  cache_key = build_view_raster_cache_key(view_features, config.export_config.content_hash())

  return ViewRasterSignature(
    schema                   = "view_raster_sig.v2",
    view_id                  = view.Id.IntegerValue,
    view_name                = view.Name,
    cache_key                = cache_key,
    build_time_utc           = utc_now(),
    descriptor_vector        = D,
    descriptor_vector_mirror = Dm,
    descriptor_kind          = config.image_normalize.descriptor_kind,
    image_kind               = config.preprocess.mode,
    crop_strategy            = config.render.crop_strategy,
    visibility_profile       = config.render.visibility_profile,
    output_px                = config.image_normalize.output_px,
    export_config_hash       = config.export_config.content_hash(),
  )


function get_or_build_view_raster(project_doc, view, view_features, raster_cache, config) -> ViewRasterSignature | null:
  # Check cache first — no Revit writes if hit.
  cached = get_cached_view_raster(raster_cache, view_features, config.export_config)
  if cached is not null:
    return cached

  sig = build_view_raster_signature(project_doc, view, view_features, config)
  if sig is Failure:
    log_warning("view raster build failed for {view.Id}: {sig.reason}")
    return null

  raster_cache.put(sig.cache_key, sig)
  return sig


# ============================================================
# SECTION 16: TWO-STAGE PIPELINE INTEGRATION
# ============================================================
#
# Stage 1: Feature-based ranking (existing; unchanged)
#   Produces score_total from tokens + geometry + fine metrics.
#   Confidence tiers are calibrated against score_total only.
#   score_total is NEVER mutated by the raster stage.
#
# Stage 2: Raster re-ranking (optional; gated on cache availability)
#   Operates on an expanded candidate pool (top_n * pool_multiplier).
#   Produces score_combined = blend of score_total and score_raster.
#   score_combined drives ordering only; it is not exposed as confidence.
#   If raster build fails for any candidate, that candidate's score_total
#   is used as score_combined (clean degradation, no crash).
#
# The symbol cache (Case 1 raster) and view raster cache (Case 2) are
# independent. The symbol cache improves token quality in Stage 1 by
# recovering geometry from opaque detail items; it is not part of Stage 2.
# The view raster cache is only consulted in Stage 2.

function find_similar_views(
  query_view,
  corpus_views,
  top_n          = 5,
  symbol_cache   = null,    # optional SymbolCache; improves Stage 1 token quality
  raster_cache   = null,    # optional view raster cache; enables Stage 2
  raster_config  = null,    # required if raster_cache is provided
  project_doc    = null,    # required if raster_cache is provided
) -> list[MatchResult]:

  # --- Stage 1: feature-based ranking (always runs) ---

  query_features = extract_features(query_view, symbol_cache=symbol_cache)
  corpus_feat    = [
    extract_features(v, symbol_cache=symbol_cache)
    for v in corpus_views
    if is_view(v) and v.Id != query_view.Id
  ]

  token_df, doc_count = build_token_df_from_features(corpus_feat)
  token_idf           = build_token_idf(token_df, doc_count)
  default_idf         = default_idf_for_doc_count(doc_count)

  results = []
  for candidate in corpus_feat:
    weights = effective_weights(query_features, candidate)
    s_tokens = token_similarity(query_features.tokens, candidate.tokens,
                                token_idf=token_idf, default_idf=default_idf)
    s_geom   = cosine_similarity(query_features.geom_fingerprint, candidate.geom_fingerprint)
    s_fine   = fine_similarity(query_features.fine_metrics, candidate.fine_metrics)
    s_total  = (weights.w_tokens * s_tokens
              + weights.w_geom   * s_geom
              + weights.w_fine   * s_fine)

    results.append(MatchResult(
      candidate_view_id = candidate.view_id,
      score_total       = s_total,        # calibrated feature score; never modified below
      score_tokens      = s_tokens,
      score_geom        = s_geom,
      score_fine        = s_fine,
      score_raster      = null,           # populated in Stage 2 if available
      score_combined    = s_total,        # default to score_total; updated in Stage 2
      confidence_tier   = confidence_tier(s_total),
      explanation       = explain_match(query_features, candidate),
    ))

  results.sort(key=(-score_total, candidate_view_id))

  # --- Stage 2: raster re-ranking (optional; skipped if cache unavailable) ---

  if raster_cache is null or raster_config is null or project_doc is null:
    # Raster layer not configured. Return feature scores as-is.
    # confidence_tier remains valid; score_total is the sole ranking signal.
    return results[0:top_n]

  pool_size     = top_n * CONFIG.rerank.pool_multiplier
  pool          = results[0:pool_size]
  tail          = results[pool_size:]

  query_raster = get_or_build_view_raster(
    project_doc, query_view, query_features, raster_cache, raster_config
  )

  if query_raster is null:
    # Query raster failed. Degrade: return Stage 1 ranking.
    log_warning("raster build failed for query view {query_view.Id}; using feature ranking")
    return results[0:top_n]

  # Build or retrieve raster for each candidate in the pool.
  view_by_id = index_by_id(corpus_views)

  for result in pool:
    candidate_view = view_by_id.get(result.candidate_view_id)
    if candidate_view is null:
      continue   # score_raster stays null; score_combined stays score_total

    candidate_raster = get_or_build_view_raster(
      project_doc,
      candidate_view,
      find_features_for_view_id(corpus_feat, result.candidate_view_id),
      raster_cache,
      raster_config,
    )

    if candidate_raster is not null:
      result.score_raster = descriptor_similarity(
        query_raster.descriptor_vector,
        query_raster.descriptor_vector_mirror,
        candidate_raster.descriptor_vector,
        candidate_raster.descriptor_vector_mirror,
      )
      result.score_combined = (
        (1.0 - CONFIG.rerank.raster_rerank_weight) * result.score_total
        + CONFIG.rerank.raster_rerank_weight        * result.score_raster
      )
    # else: score_combined stays score_total (clean degradation per candidate)

  pool.sort(key=(-score_combined, candidate_view_id))

  # Append non-pool tail unchanged (no raster score; not re-ranked).
  return (pool + tail)[0:top_n]


# ============================================================
# SECTION 17: CACHE STORAGE AND LIFECYCLE
# ============================================================
#
# Storage layout:
#
#   {user_docs}/revit_detail_intelligence/cache/
#     symbols/
#       {project_slug}.symbols.json    # per-project; shareable if same family library
#     views/
#       {project_slug}.views.json      # always project-scoped; view IDs are local
#
# Symbol cache sharing:
#   The symbol cache key is name-based (family|type), so a descriptor built
#   on project A is valid for project B if the same family is loaded and the
#   validity token matches. Sharing is opt-in: point both projects at the same
#   cache file. Not recommended if family versions differ between projects.
#
# View raster cache:
#   Always project-scoped. The cache key includes view_id, which is not
#   stable across projects.
#
# Pruning:
#   Symbol cache: no time expiry; entries are small and invalidate on content change.
#   View raster cache: prune entries older than max_age_days on load;
#                      evict LRU entries beyond max_entries.
#
# Manual invalidation:
#   Delete the JSON file to force a full rebuild.
#   Graphical changes that don't affect parameters or geometry (line weight,
#   fill pattern) will not auto-invalidate; manual deletion is the escape hatch.

function load_or_create_symbol_cache(path) -> SymbolCache:
  if file_exists(path):
    cache = read_json(path)
    if cache.schema != "symbol_cache.v2":
      log_info("symbol cache schema mismatch at {path}; starting fresh")
      return empty_symbol_cache()
    return cache
  return empty_symbol_cache()


function load_or_create_view_raster_cache(path, policy) -> ViewRasterCache:
  if file_exists(path):
    cache = read_json(path)
    if cache.schema != "view_raster_cache.v2":
      log_info("view raster cache schema mismatch; starting fresh")
      return empty_view_raster_cache()
    cache = prune_expired_entries(cache, policy.max_age_days)
    cache = enforce_max_entries(cache, policy.max_entries, evict="lru")
    return cache
  return empty_view_raster_cache()
